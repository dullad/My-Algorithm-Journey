```c++
class Solution {
    int my_search(vector<int>& nums, int target, int left, int right){
        // 有序数组找target
        if(nums[right] < target || nums[left] > target) return -1;
        while(left <= right){
            int mid = (left + right) / 2;
            if(nums[mid] == target) return mid;
            else if(nums[mid] < target){
                left = mid + 1;
            }else{
                right = mid - 1;
            }
        }
        return -1;
    }
public:
    int search(vector<int>& nums, int target) {
        int left = 0, right = nums.size() - 1;
        while(left <= right){
            int mid = (left + right) / 2;
            if(nums[mid] == target) return mid;
            else{
                if(nums[mid] >= nums[left]){// 注意，每次找左边序列的时候，把mid包含进去
                    // 左边有序
                    int pos = my_search(nums, target, left, mid);
                    if(pos != -1) return pos;
                    else{ // 有序区间没找到，继续去另一侧找
                        left = mid + 1;
                    }
                }
                else{
                    // 右边有序
                    int pos = my_search(nums, target, mid + 1, right);
                    if(pos != -1) return pos;
                    else{
                        right = mid;
                    }
                }
            }
        }
        return -1;
    }
};
```

### 4. 复杂度分析

- **时间复杂度**: $O(N)$。我们只遍历数组一次。对于每次遍历，哈希表（`unordered_map`）的插入和查找操作的平均时间复杂度都是 $O(1)$。因此，总时间复杂度是线性的。
- **空间复杂度**: $O(N)$。在最坏的情况下，我们需要将数组中的 N 个元素都存入哈希表中（例如，没有找到解，或者解在最后两个元素）。因此，哈希表所需空间与数组大小成正比。

### 5. 总结与反思

- **问题转换**：解题的关键在于将问题 “寻找 `a` 和 `b` 使 `a + b = target`” 转换为 “对于每个元素 `x`，寻找 `target - x` 是否已经存在”。
- **哈希表的妙用**：哈希表提供了近乎 $O(1)$ 的查找效率，是解决这类 “检查元素是否存在” 问题的首选数据结构。
